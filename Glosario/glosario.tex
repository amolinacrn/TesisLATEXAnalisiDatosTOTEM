\begin{eqlist}[\def\makelabel#1{\hfill#1}\labelsep 1em]% NO BORRAR
\item[\bf Bos\'on:] Partícula con espín entero, portadora de una de las cuatro interacciones fundamentales de la materia.
\item[\bf Quark:] Constituyentes fundamentales de la materia. Poseen espín semi-entero.
\item[\bf Diquark:]  Cuasi-partícula, estado de dos quarks
\item[\bf CERN:] Centro Europeo  para la investigación nuclear
\item[\bf Fermi\'on:] Partícula con sp\'in semi-entero
\item[\bf Hadr\'on:] Part\'icula subat\'omica que se compone de dos o tres quarks.
 \item[\bf Fot\'on:] partícula mediadora (de la fuerza) en las interacciones electromagnéticas.
 \item[\bf Glu\'on:] partícula mediadora (de la fuerza) en las interacciones fuertes.
 \item[\bf Cromodin\'amica cu\'antica (QCD):] Teoría cuántica de campos que describe las iteraciones fuertes.
 \item[\bf Pomer\'on:] Trayectoria de Regge con intercepto mayor a 1.
 \item[\bf Polos:] Singularidades en el pano complejo.
 \item[\bf Dispersión:] Alteración en la trayectoria de una partícula como consecuencia de una interacción .
